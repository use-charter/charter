Design the script:

```typescript
// scripts/generate-rule-pages.ts
// Generates docs/product/rules/AE-*.mdx from docs/internal/specs/AE-*.md.
// Run: bun scripts/generate-rule-pages.ts
// With --check: exit non-zero if any rule is missing or drifted.

import { readdirSync, readFileSync, writeFileSync, existsSync } from 'node:fs';
import { resolve } from 'node:path';

const REPO_ROOT = resolve(import.meta.dirname, '..');
const SPECS_DIR = resolve(REPO_ROOT, 'docs/internal/specs');
const RULES_DIR = resolve(REPO_ROOT, 'docs/product/rules');

// Parse structured sections from spec markdown.
// Each spec has sections: ## Rule ID, ## Severity, ## Category, ## Description,
// ## Detection logic, ## Pass example, ## Fail example, ## Remediation, etc.
interface RuleSpec {
  id: string;
  severity: string;
  category: string;
  description: string;
  detection: string;
  passExample: string;
  failExample: string;
  remediation: string;
  scoringImpact: string;
  relatedAdrs: string[];
  edgeCases: string;
}

function parseSpec(filePath: string): RuleSpec | null {
  const content = readFileSync(filePath, 'utf-8');
  // Extract sections by ## headers
  const sections = new Map<string, string>();
  let currentHeader = '';
  for (const line of content.split('\n')) {
    const h = line.match(/^## (.+)/);
    if (h) {
      currentHeader = h[1].toLowerCase().replace(/[^a-z0-9 ]/g, '').trim();
      sections.set(currentHeader, '');
    } else if (currentHeader) {
      sections.set(currentHeader, (sections.get(currentHeader) || '') + line + '\n');
    }
  }

  const id = filePath.match(/AE-[A-Z]+-\d+/)?.[0] || '';
  if (!id) return null;

  return {
    id,
    severity: extractValue(sections.get('severity') || '', 'Severity'),
    category: extractValue(sections.get('category') || '', 'Category'),
    description: sections.get('description') || '',
    detection: sections.get('detection logic') || '',
    passExample: sections.get('pass example') || '',
    failExample: sections.get('fail example') || '',
    remediation: sections.get('remediation') || '',
    scoringImpact: sections.get('scoring impact') || '',
    edgeCases: sections.get('edge cases') || '',
    relatedAdrs: extractList(sections.get('related adrs') || ''),
  };
}

function extractValue(text: string, label: string): string {
  const lines = text.trim().split('\n');
  for (const line of lines) {
    const trimmed = line.trim();
    if (trimmed.startsWith('-')) return trimmed.replace(/^- /, '').trim();
  }
  return '';
}

function extractList(text: string): string[] {
  return text.split('\n')
    .map(l => l.trim())
    .filter(l => l.startsWith('-'))
    .map(l => l.replace(/^- /, '').trim());
}

function toMdx(spec: RuleSpec): string {
  const severityTag = spec.severity.toLowerCase();
  return `---
title: "${spec.id} — ${spec.severity}"
description: "${spec.description.slice(0, 120).replace(/"/g, '&quot;')}"
---

<Info>**Severity:** ${spec.severity}  ·  **Category:** ${spec.category}</Info>

${spec.description}

---

## Detection

${spec.detection}

${spec.passExample ? `## Pass Example\n\n${spec.passExample}\n` : ''}
${spec.failExample ? `## Fail Example\n\n${spec.failExample}\n` : ''}
${spec.edgeCases ? `## Edge Cases\n\n${spec.edgeCases}\n` : ''}

${spec.remediation ? `## Remediation\n\n${spec.remediation}\n` : ''}

${spec.scoringImpact ? `## Scoring Impact\n\n${spec.scoringImpact}\n` : ''}

${spec.relatedAdrs.length ? `## Related ADRs\n\n${spec.relatedAdrs.map(a => `- ${a}`).join('\n')}\n` : ''}

<Card title="charter explain ${spec.id}" icon="terminal" href="/cli/explain">
  Run \`charter explain ${spec.id}\` to see this rule's rationale and remediation from the CLI.
</Card>
`;
}

function main() {
  const check = process.argv.includes('--check');
  const specFiles = readdirSync(SPECS_DIR).filter(f => f.startsWith('AE-') && f.endsWith('.md'));
  let written = 0;
  let errors = 0;

  for (const sf of specFiles) {
    const spec = parseSpec(resolve(SPECS_DIR, sf));
    if (!spec) { errors++; continue; }
    const outPath = resolve(RULES_DIR, spec.id + '.mdx');
    const mdx = toMdx(spec);
    if (check) {
      if (!existsSync(outPath)) {
        console.error(`Missing: ${outPath}`);
        errors++;
      }
    } else {
      writeFileSync(outPath, mdx);
      written++;
    }
  }

  // Validate coverage: every catalog helpUri should have a page
  // Read catalog.go and extract rule IDs
  const catalogPath = resolve(REPO_ROOT, 'internal/rules/catalog/catalog.go');
  const catalogContent = readFileSync(catalogPath, 'utf-8');
  const catalogIds = [...catalogContent.matchAll(/help\("(AE-[A-Z]+-\d+)"\)/g)].map(m => m[1]);

  for (const id of catalogIds) {
    if (!existsSync(resolve(RULES_DIR, id + '.mdx'))) {
      console.error(`Missing rule page for catalog entry: ${id}`);
      errors++;
    }
  }

  if (check) {
    if (errors > 0) process.exit(1);
    console.log(`Rule pages: PASS — all ${catalogIds.length} pages present.`);
  } else {
    console.log(`Generated ${written} rule pages.`);
  }
}

main();
```

```typescript
// scripts/validate-product-docs.ts
// Validates the Mintlify product docs project.
// Checks: docs.json exists, navigation entries resolve to real files,
// all rule pages present, moon run :docs is green.

import { readFileSync, existsSync } from 'node:fs';
import { resolve } from 'node:path';

const REPO_ROOT = resolve(import.meta.dirname, '..');
const PRODUCT_DIR = resolve(REPO_ROOT, 'docs/product');

const errors: string[] = [];

// docs.json must exist
const docsJsonPath = resolve(PRODUCT_DIR, 'docs.json');
if (!existsSync(docsJsonPath)) {
  errors.push('docs.json missing from docs/product/');
}

// All navigation page entries should correspond to existing MDX files
const docsJson = JSON.parse(readFileSync(docsJsonPath, 'utf-8'));
function collectPages(node: any): string[] {
  if (typeof node === 'string') return [node];
  if (Array.isArray(node)) return node.flatMap(collectPages);
  if (node && typeof node === 'object') {
    return collectPages(Object.values(node));
  }
  return [];
}
const pages = collectPages(docsJson.navigation);
for (const page of pages) {
  if (page.startsWith('http')) continue;
  const mdxPath = resolve(PRODUCT_DIR, page + '.mdx');
  if (!existsSync(mdxPath)) {
    errors.push(`Navigation entry "${page}" has no corresponding MDX file`);
  }
}

if (errors.length > 0) {
  console.error('Product docs validation FAILED:');
  for (const e of errors) console.error(`  ✗ ${e}`);
  process.exit(1);
}
console.log(`Product docs: PASS (${pages.length} navigation entries resolved).`);
```
