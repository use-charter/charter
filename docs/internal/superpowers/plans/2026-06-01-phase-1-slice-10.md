package perf

import (
	"fmt"
	"os"
	"path/filepath"
	"runtime"
	"testing"
	"time"

	"go.use-charter.dev/charter/internal/doctor"
)

	fileCount    = 50000
	maxWall      = 2 * time.Second
	maxRSSBytes  = 256 * 1024 * 1024
)

// buildLargeRepo synthesizes a representative ~50k-file repository in a temp dir.
// Match the form the inventory scanner consumes (see Step 1): if it shells to
// `git ls-files`, git-init and add the files; otherwise a plain tree is enough.
func buildLargeRepo(t *testing.T) string { /* implement per Step 1 */ return "" }

func TestDoctorPerformance(t *testing.T) {
	root := buildLargeRepo(t)

	start := time.Now()
	if _, err := doctor.Run(root, 80, true); err != nil {
		t.Fatalf("doctor.Run: %v", err)
	}
	elapsed := time.Since(start)
	if elapsed > maxWall {
		t.Fatalf("scan took %s, want <= %s", elapsed, maxWall)
	}
	t.Logf("scan of %d files: %s", fileCount, elapsed)

	if runtime.GOOS == "linux" {
		rss, err := peakRSSBytes()
		if err != nil {
			t.Fatalf("read peak RSS: %v", err)
		}
		t.Logf("peak RSS: %d MiB", rss/1024/1024)
		if rss > maxRSSBytes {
			t.Fatalf("peak RSS %d bytes, want <= %d", rss, maxRSSBytes)
		}
	} else {
		var m runtime.MemStats
		runtime.ReadMemStats(&m)
		t.Logf("non-linux: HeapSys=%d MiB (RSS assertion skipped)", m.HeapSys/1024/1024)
	}
}

// peakRSSBytes reads VmHWM (peak resident set size) from /proc/self/status (Linux).
func peakRSSBytes() (uint64, error) {
	data, err := os.ReadFile("/proc/self/status")
	if err != nil {
		return 0, err
	}
	for _, line := range splitLines(string(data)) {
		var kb uint64
		if n, _ := fmt.Sscanf(line, "VmHWM: %d kB", &kb); n == 1 {
			return kb * 1024, nil
		}
	}
	return 0, fmt.Errorf("VmHWM not found in /proc/self/status")
}

func splitLines(s string) []string { /* strings.Split(s, "\n") */ return nil }

## Task 2: The composite Action (`action/action.yml` + README)

**Files:** create `action/action.yml`, rewrite `action/README.md`. (Verify asset names against Slice 9: snapshot produced `charter_<version>_<os>_<arch>.{tar.gz,zip}` + `checksums.txt`.)

- [ ] **Step 1: Write `action/action.yml`** (composite). Use this as the reference; verify (a) the SHA pins for `sigstore/cosign-installer` (pin `cosign-release: v3.0.6`) and `github/codeql-action/upload-sarif` (**v4**) — look up current full SHAs; (b) `.Version` in the asset name is the tag **without** the leading `v` (GoReleaser strips it); (c) checksum verification is portable across Linux (`sha256sum`) and macOS (`shasum -a 256`).

```yaml
name: "Charter"
description: "Scan a repository for AI-agent readiness and upload SARIF to GitHub Code Scanning."
author: "use-charter"
branding:
  icon: "shield"
  color: "purple"

inputs:
  version:
    description: "Charter release to use (e.g. v1.0.0), or 'latest'."
    required: false
    default: "latest"
  path:
    description: "Repository path to scan."
    required: false
    default: "."
  threshold:
    description: "Minimum passing score (0-100). Empty defers to charter.yaml policy / default 80."
    required: false
    default: ""
  format:
    description: "Output format: text, json, markdown, or sarif."
    required: false
    default: "sarif"
  sarif-file:
    description: "Path to write the SARIF report."
    required: false
    default: "charter.sarif"
  upload:
    description: "Upload SARIF to GitHub Code Scanning (requires security-events: write)."
    required: false
    default: "true"
  verify:
    description: "Verify the binary with cosign + sha256 before running."
    required: false
    default: "true"
  fail-below:
    description: "Fail the step when the score is below the threshold."
    required: false
    default: "true"
  category:
    description: "Optional Code Scanning category for the SARIF results."
    required: false
    default: ""

outputs:
  exit-code:
    description: "charter doctor exit code (0 pass, 1 below threshold, 2 error)."
    value: ${{ steps.run.outputs.exit-code }}
  sarif-file:
    description: "Path to the written SARIF report."
    value: ${{ steps.run.outputs.sarif-file }}

runs:
  using: "composite"
  steps:
    - name: Install cosign
      if: inputs.verify == 'true'
      uses: sigstore/cosign-installer@<SHA> # v4.1.1
      with:
        cosign-release: "v3.0.6"
    - name: Download and verify Charter
      id: install
      shell: bash
      env:
        IN_VERSION: ${{ inputs.version }}
        IN_VERIFY: ${{ inputs.verify }}
        GH_TOKEN: ${{ github.token }}
      run: |
        set -euo pipefail
        ver="$IN_VERSION"
        if [ "$ver" = "latest" ]; then
          ver="$(gh release view --repo use-charter/charter --json tagName -q .tagName)"
        fi
        nover="${ver#v}"
        case "$RUNNER_OS" in
          Linux) os="linux" ;;
          macOS) os="darwin" ;;
          Windows) os="windows" ;;
          *) echo "unsupported OS: $RUNNER_OS" >&2; exit 1 ;;
        esac
        case "$RUNNER_ARCH" in
          X64) arch="amd64" ;;
          ARM64) arch="arm64" ;;
          *) echo "unsupported arch: $RUNNER_ARCH" >&2; exit 1 ;;
        esac
        ext="tar.gz"; [ "$os" = "windows" ] && ext="zip"
        base="charter_${nover}_${os}_${arch}"
        work="$(mktemp -d)"
        gh release download "$ver" --repo use-charter/charter \
          --pattern "${base}.${ext}" --pattern "checksums.txt" --pattern "checksums.txt.sigstore.json" \
          --dir "$work"
        cd "$work"
        if [ "$IN_VERIFY" = "true" ]; then
          cosign verify-blob \
            --bundle "checksums.txt.sigstore.json" \
            --certificate-identity "https://github.com/use-charter/charter/.github/workflows/release.yml@refs/tags/${ver}" \
            --certificate-oidc-issuer "https://token.actions.githubusercontent.com" \
            "checksums.txt"
          line="$(grep " ${base}.${ext}\$" checksums.txt)"
          if command -v sha256sum >/dev/null 2>&1; then echo "$line" | sha256sum -c -; else echo "$line" | shasum -a 256 -c -; fi
        fi
        if [ "$ext" = "zip" ]; then unzip -q "${base}.${ext}"; else tar -xzf "${base}.${ext}"; fi
        bin="charter"; [ "$os" = "windows" ] && bin="charter.exe"
        echo "charter-bin=${work}/${bin}" >> "$GITHUB_OUTPUT"
    - name: Run charter doctor
      id: run
      shell: bash
      env:
        CHARTER_BIN: ${{ steps.install.outputs.charter-bin }}
        IN_PATH: ${{ inputs.path }}
        IN_FORMAT: ${{ inputs.format }}
        IN_SARIF: ${{ inputs.sarif-file }}
        IN_THRESHOLD: ${{ inputs.threshold }}
      run: |
        set -uo pipefail
        args=(doctor --path "$IN_PATH" --format "$IN_FORMAT" --out "$IN_SARIF")
        [ -n "$IN_THRESHOLD" ] && args+=(--threshold "$IN_THRESHOLD")
        "$CHARTER_BIN" "${args[@]}"; code=$?
        echo "exit-code=${code}" >> "$GITHUB_OUTPUT"
        echo "sarif-file=${IN_SARIF}" >> "$GITHUB_OUTPUT"
    - name: Upload SARIF
      if: inputs.upload == 'true' && inputs.format == 'sarif'
      uses: github/codeql-action/upload-sarif@<SHA> # v4
      with:
        sarif_file: ${{ inputs.sarif-file }}
        category: ${{ inputs.category }}
    - name: Apply gate
      shell: bash
      env:
        CODE: ${{ steps.run.outputs.exit-code }}
        FAIL_BELOW: ${{ inputs.fail-below }}
      run: |
        set -euo pipefail
        [ "$CODE" = "2" ] && { echo "charter: scan error" >&2; exit 2; } || true
        [ "$CODE" = "1" ] && [ "$FAIL_BELOW" = "true" ] && { echo "charter: score below threshold" >&2; exit 1; } || true
        exit 0
```

## Decisions (locked in ADR-0018)

1. Composite action in `action/` (source of truth) → seeded to dedicated `use-charter/charter-action@v1` at launch; Marketplace-ready.
2. Flow: resolve version → download → cosign `verify-blob` (release.yml identity) + sha256 → run `doctor --format sarif --out` → `upload-sarif@v4` → propagate exit (upload before gate).
3. Inputs with escape hatches (`verify:false`, `upload:false`); full verification + auto-upload by default.
4. All third-party `uses:` SHA-pinned; a repo validator enforces it on `action.yml` (zizmor/actionlint don't scan it) in `:check`.
5. Armed-not-fired: validate structure + perf now; first e2e at release. Charter CI keeps `go run ./cmd/charter doctor`.
6. Perf: synthesized 50k-file repo, ≤ 2 s wall-clock + Linux `VmHWM` ≤ 256 MB, build-tagged `:perf` (not in `:check`).

A change to any of these during execution updates the ADR (Task 4 Step 1), not just the plan.
